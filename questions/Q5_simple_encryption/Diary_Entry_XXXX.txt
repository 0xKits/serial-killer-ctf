def encrypt(text):
    primes = get_first_n_primes(len(text))
    shift = len(text)
    rotated_primes = primes[shift:] + primes[:shift]
    encrypted = [chr((ord(c)<<2) ^ rotated_primes[i]) for i, c in enumerate(text)]
    return "".join(encrypted)
